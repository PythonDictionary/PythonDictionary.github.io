entry

import tkinter as tk
    
def get_entry_value():
    text = entry.get() # Get text from entry widget
    print("Text from entry:", text) # Do something with the value

root = tk.Tk()
canvas = tk.Canvas(root, width=200, height=200)
canvas.pack()

# Create entry widget and place it on canvas
entry = tk.Entry(canvas)
entry.place(x=65, y=80)

# Create button and place it on canvas
button = tk.Button(canvas, text="Get Text", command=get_entry_value)
button.place(x=65, y=100)

root.mainloop()
