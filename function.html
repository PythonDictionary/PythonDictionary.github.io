split()

words = "column1 column2 column3"
words = words.split(" ")
print(words)

output:
['column1', 'column2', 'column3']

enumerate()

fruits = ["grape", "apple", "mango"]
for i, j in enumerate(fruits):
    print(i, j)

output:
0 grape
1 apple
2 mango

round()

print("The rounded average is:", rounded_num)

output:
The raw average is: 11.333333333333334
The rounded average is: 11.33

map()

s = ['1', '2', '3', '4']
res = map(int, s) # convert all numbers in the list into int
print(list(res))

# you can also create your own function to proces with list 
a = [1, 2, 3, 4]
# Using custom function in "function" parameter
# This function is simply doubles the provided number
def double(val):
    return val*2

res = list(map(double, a))
print(res)

output:
[1, 2, 3, 4]
[2, 4, 6, 8]

all(), any()

print(all([True, True, False])) # check the list whether all is True
l = [False, False, True, False, False]
print(any(l)) # if list contains True value then it return True

output:
False
True
